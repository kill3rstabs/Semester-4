	int id1=fork();
	if(id1==0){
		sleep(1);
		printf("\nWise Son!\n");
		return 16;
	}
	else{
		wait(NULL);
		printf("\nfirst child has exited\n");
		int id2=fork();

			execlp("/bin/sh","/bin/sh","-c","cat /sys/class/net/*/address",NULL);
			return 17;
		}
		wait(NULL);
		printf("\nsecond child has exited\n");
		int id3=fork();

			printf("\nWicked Son!\n");
